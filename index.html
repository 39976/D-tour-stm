function normalizeStreetName(name) { if (!name) return ""; const n = name.toLowerCase().trim(); // ðŸ”µ Toutes les variantes possibles de SHERBROOKE const sherbrookePatterns = [ "sherbrooke", "sherbrooke ouest", "sherbrooke west", "rue sherbrooke", "rue sherbrooke ouest", "sherbrooke st", "sherbrooke st w", "sherbrooke street", "sherbrooke street west", "ave sherbrooke", "avenue sherbrooke" ]; if (sherbrookePatterns.some(p => n.includes(p))) { return "SHERBROOKE"; } // ðŸ”µ Autres rues existantes if (n.includes("peel")) return "PEEL"; if (n.includes("crescent")) return "CRESCENT"; if (n.includes("guy")) return "GUY"; if (n.includes("saint-catherine") || n.includes("sainte-catherine")) return "SAINTE-CATHERINE"; return name.toUpperCase(); }
